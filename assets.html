					<a href="#modal-content-1" class="fancybox"> CLICK this to see MODAL! </a>

					// After creating your button, or wrapping content in the fancybox 'a' tag to enable the
					// modal, create a div with the ID you targeted with your href in the fancybox 'a' tag 
					// and a class of modal-window that applies the correct css to your modal
					<div id="modal-content-1" class="modal-window">
